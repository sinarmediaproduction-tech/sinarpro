// ============ ARTAFLOW BACKEND SCRIPT ============ const SHEET_NAME = "Transactions"; function doGet() { return handleRequest(); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(["id", "date", "desc", "type", "amount", "created_at"]); } const data = e ? JSON.parse(e.postData.contents) : {}; const action = data.action || "get"; if (action === "add" && data.transaction) { const t = data.transaction; sheet.appendRow([t.id, t.date, t.desc, t.type, t.amount, new Date().toISOString()]); return ContentService.createTextOutput(JSON.stringify({ success: true })); } if (action === "update" && data.id && data.transaction) { const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] == data.id) { sheet.getRange(i + 1, 1, 1, 5).setValues([[ data.transaction.id, data.transaction.date, data.transaction.desc, data.transaction.type, data.transaction.amount ]]); break; } } return ContentService.createTextOutput(JSON.stringify({ success: true })); } if (action === "delete" && data.id) { const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] == data.id) { sheet.deleteRow(i + 1); break; } } return ContentService.createTextOutput(JSON.stringify({ success: true })); } if (action === "sync" && data.transactions) { sheet.clear(); sheet.appendRow(["id", "date", "desc", "type", "amount", "created_at"]); data.transactions.forEach(t => { sheet.appendRow([t.id, t.date, t.desc, t.type, t.amount, t.created_at || new Date().toISOString()]); }); return ContentService.createTextOutput(JSON.stringify({ success: true })); } const rows = sheet.getDataRange().getValues(); const transactions = []; for (let i = 1; i < rows.length; i++) { transactions.push({ id: rows[i][0], date: rows[i][1], desc: rows[i][2], type: rows[i][3], amount: rows[i][4], created_at: rows[i][5] }); } return ContentService.createTextOutput(JSON.stringify({ success: true, transactions: transactions })); } catch(error) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString() })); } }
